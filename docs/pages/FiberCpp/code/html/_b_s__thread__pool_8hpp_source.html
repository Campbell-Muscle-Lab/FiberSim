622 void worker()
623 {
624 std::function<void()> task;
625 while (true)
626 {
627 std::unique_lock tasks_lock(tasks_mutex);
628 task_available_cv.wait(tasks_lock, [this] { return !tasks.empty() || !workers_running; });
629 if (!workers_running)
630 break;
631 if (paused)
632 continue;
633 task = std::move(tasks.front());
634 tasks.pop();
635 ++tasks_running;
636 tasks_lock.unlock();
637 task();
638 tasks_lock.lock();
639 --tasks_running;
640 if (waiting && !tasks_running && (paused || tasks.empty()))
641 tasks_done_cv.notify_all();
642 }
643 }
